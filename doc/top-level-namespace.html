post('/save_listing') do
  listing_id = params[:listing_id]
  user_id = session[:id]

  if user_id.nil?
    flash[:user_guest] = "Du måste vara inloggad för denna tjänst."
    redirect('/listing/index')
  else
    save_listing(listing_id, user_id)
    redirect('/listing/index')
  end
end
